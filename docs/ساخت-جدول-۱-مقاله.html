column <- "Degree"

value_counter <- function(data, column) {
  values <- unique(data[, column])
  n_values <- length(values)
  df <- data.frame()
  
  for (i in seq_len(n_values)) {
    df[i, "value"] <- values[i]
    df[i, "n"] <- sum(data[, column] == values[i])
    df[i, "p"] <- 
      (sum(data[, column] == values[i]) / nrow(survey) * 100) |> round(1)
  }
  
  df
}

Code 
columns <- c("Degree", "Gender")

values_counter <- function(data, columns) {
  n_columns <- length(columns)
  
  for (i in seq_len(n_columns)) {
    df <- data |> value_counter(columns[i])
    df$column <- columns[i]
    
    if (i == 1) {
      dfs <- df
    } else {
      dfs <- dfs |> rbind(df)
    }
  }
  
  dfs[c("column", "value", "n", "p")]
}
