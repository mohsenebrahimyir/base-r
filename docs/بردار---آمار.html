چارک اول (\({\bf Q1}\)), چارک دوم یا همان میانه (\({\bf Q2\;or\;Median}\)), چارک سوم (\({\bf Q3}\)),

\[\Large x_{sort} = \left[ \begin{array}{ccccccccccc} -7 & -2 & \overset{\overset{{\bf Q1}}{\uparrow}}{-2} & 0 & 1 & \overset{\overset{{\bf Q2}}{\uparrow}}{3} & 4 & 5 & \overset{\overset{{\bf Q3}}{\uparrow}}{5} & 5 & 6 \end{array} \right] \]

برای محاسبه چارک‌ها باید تابع جدید ساخت زیرا تابع پیش فرضی وجود ندارد

  1. تعریف تابع
Code 
quart <- function(x) {
  x <- sort(x)
  n <- length(x)
  m <- (n+1)/2
  if (floor(m) != m) {
    l <- m-1/2; u <- m+1/2
  } else {
    l <- m-1; u <- m+1
  }
  c(Q1=median(x[1:l]), Q2=median(x), Q3=median(x[u:n]))
}
  1. استفاده از تابع
Code 
quart(stat)
## Q1 Q2 Q3 
## -2  3  5

modes <- function(x) {
  ux <- unique(x)
  tab <- tabulate(match(x, ux))
  ux[tab == max(tab)]
}
